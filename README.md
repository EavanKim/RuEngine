# RuEngine
 RuEngine

(type)*     =   p(typeNick)
types
char        =   sz
int         =   i
float       =   f
double      =   lf
short       =   si
byte        =   bt
struct      =   tag

unsigned    =   u

class member value      =   m(type)_(name)
param value             =   _(name)
templary value          =   tmp_(name)

increase index          =   i(type)(name)

thread function         =   thread_(name)
